        <script>

                const form = document.querySelector('form');

                form.addEventListener('submit', (event) => {

                        event.preventDefault();

                        const number = document.getElementById('numbers').value;

                        const luckyNumber = getLuckyNumber(number);

                        document.getElementById('output').innerText = `Lucky number is ${luckyNumber} for number ${number}.`;

                });

                function getLuckyNumber(number) {

                        // This is an example CSV file with the lucky numbers for the 12 numbers

                        const csv = `Number,Lucky Number

PRASAD,GHANTASALA

2,3

3,9

4,2

5,11

6,8

7,1

8,10

9,4

10,12

11,6

12,5`;

                        // Parse the CSV file into an array of objects

                        const lines = csv.split('\n');

                        const headers = lines[0].split(',');

                        const data = lines.slice(1).map(line => {

                                const values = line.split(',');

                                return headers.reduce((obj, header, index) => {

                                        obj[header] = values[index];

                                        return obj;

                                }, {});

                        });

                        // Find the lucky number for the given number

                        const record = data.find(record => record.Number === number);

                        if (record) {

                                return record['Lucky Number'];

                        } else {

                                return -1; // if the number is not found, return -1

                        }

                }

        </script>
